# 发送邮件

  1. 后台配置邮件参数
  2. 引用命名空间 use sunphp\email\SunEmail
  3. 调用send()方法既可
use sunphp\email\SunEmail;

/* 前端默认入口示例 */
class Email {


    public function send(){

        // 图片上传
        $params=[
            'email'=>'1457466046@qq.com',//用户邮箱
            'name'=>'用户昵称',
            'title'=>'邮件标题',
            'content'=>'邮件正文内容'
        ];
		$res = SunEmail::send($params);
        if($res){
            echo "邮件发送成功";
        }
    }

}
