# 获取openid信息

  1. 引入命名空间 use sunphp\account\SunAccount;
  2. 调用session()方法既可
    /* 获取用户的openid */
    public function getOpenid(){
        $code="小程序前端获取的code";
        $account=SunAccount::create();
        $data=$account->session($code);

        //获取到的数据格式如下
        // $data=[
        //     'openid'=>"获取到的用户openid",
        //     'session_key'=>"获取到的session值",
        // ];
    }
